Array2d U() {
  Array2d U = Array2d.fixed(_n, _n);
  for (int i = 0; i < _n; i++) {
    for (int j = 0; j < _n; j++) {
      if (i <= j) {
        U[i][j] = _LUMatrix[i][j];
      } else {
        U[i][j] = 0.0;
      }
    }
  }
  return U;
}
